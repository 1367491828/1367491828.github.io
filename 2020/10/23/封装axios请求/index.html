封装axios请求 

import axios from 'axios';
import { MessageBox, Message } from 'element-ui'
// 使用element ui

​
// 创建axios实例，
const instance = axios.create({
// baseURL: process.env.NODE_ENV !== ‘production’ ? ‘/api’ : process.env.VUE_APP_BASE_API,
//baseURL根据运行环境来设置baseUrl，会自动拼接api接口中的路径，当环境为开发环境时，会发生开发跨域，所以需要在vue.config.js中配置
// 跨域代理，会对/api开头的api进行代理
// baseURL: ‘https://mock.cangdu.org/mock/5f9276a43651b04f6852270a/example',
timeout: 5000,
})

​
// 请求拦截器
instance.interceptors.request.use(function(config) {
// config.headers[‘Authorization’] = localStorage.getItem(‘token’)
// 从本地缓存中取出token,其他方法也可以

    return config
}, function(error) {

    return Promise.reject("请求配置发生错误!")
})

​

​
// 响应拦截器
instance.interceptors.response.use(function(response) {
// 对请求成功的数据进行一些处理
return response;

}, function(error) {
let message;
if (error.response.status) {

    let status = error.response.status

    if (status === 403) {
        message = '验证过期，你需要重新登录才能继续操作！'
        // MessageBox.confirm(res, '确认退出', {
        //     confirmButtonText: '重新登录',
        //     cancelButtonText: '取消',
        //     type: 'warning'
        // }).then(() => {
        //     store.dispatch('user/resetToken').then(() => {
        //         location.reload();
        //     })
        // })   //对于过期需要对token进行重新设置
        return new Promise(()=>{}); //中断Promise链才不会被请求的catch再捕获处理
    }

    message = getMessage(status)

} else {

    message = "网络断开或服务器错误！";

}

return Promise.reject(message);

})

​
function getMessage(status) {
let res;
switch (status) {
case 401:
res = “未授权请先登录！”;
break;
case 404:
res = “资源找不到了！”;
break;
case 500:
res = “服务器走小差了！”;
break;
default:
res = “服务器走小差了！”
}
return res
}

export default instance;
